What is the output of this functional component?

Counter.jsx 
import { useState, useEffect } from 'react';

export default function Counter() {
  const [count, setCount] = useState(0);

  useEffect(() => {
    let timerId;
    if (count < 5) {
      timerId = setTimeout(() => {
        setCount(c => c + 1);
      }, 1000);
    }
    
    // Cleanup function
    return () => clearTimeout(timerId);
  }, [count]);

  return (
    <div>
      <p>Count: {count}</p>
    </div>
  );
}
